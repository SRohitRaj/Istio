set -e
set -u
set -o pipefail

# $snippet test_fortio_httpbin_interaction.sh syntax="bash"
$ FORTIO_POD=$(kubectl get pods -n istio-io-circuitbreaker | grep fortio | awk '{ print $1 }')
$ kubectl -n istio-io-circuitbreaker exec -it $FORTIO_POD -c fortio /usr/bin/fortio -- load -curl http://httpbin:8000/get
# $verify verifier="contains"
200 OK
# $endsnippet

# $snippet almost_trip_circuit_breaker.sh syntax="bash"
$ kubectl -n istio-io-circuitbreaker exec -it $FORTIO_POD -c fortio /usr/bin/fortio -- load -c 2 -qps 0 -n 20 -loglevel warning http://httpbin:8000/get
# $endsnippet

# $snippet trip_circuit_breaker.sh syntax="bash"
$ kubectl -n istio-io-circuitbreaker exec -it $FORTIO_POD -c fortio /usr/bin/fortio -- load -c 3 -qps 0 -n 30 -loglevel warning http://httpbin:8000/get
# $endsnippet

# $snippet print_statistics_after_tripping.sh syntax="bash"
$ kubectl -n istio-io-circuitbreaker exec $FORTIO_POD -c istio-proxy -- pilot-agent request GET stats | grep httpbin | grep pending
# $endsnippet
